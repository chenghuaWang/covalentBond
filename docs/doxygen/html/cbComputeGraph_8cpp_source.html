348 WFGraphTask* cbComputeGraph::generateGraphTask(const graph_callback& func) {
349  WFGraphTask* graph = WFTaskFactory::create_graph_task([=](WFGraphTask* task) {
350  fmt::print(fg(fmt::color::steel_blue) | fmt::emphasis::italic,
