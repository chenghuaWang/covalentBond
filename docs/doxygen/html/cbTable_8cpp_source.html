141 std::map<int32_t, int32_t> cbVirtualTable::keyBy(const std::string& colName) const {
142  std::map<int32_t, int32_t> tmp;
143  int32_t row = m_shape[0];
144  int32_t col = m_shape[1];
145  int atCol = 0;
146  // hash funcs
147  std::hash<std::string> hash_string;
148  [[maybe_unused]] std::hash<float> hash_float;
149  [[maybe_unused]] std::hash<double> hash_double;
150  [[maybe_unused]] std::hash<int> hash_int;
151  [[maybe_unused]] std::hash<unsigned long long> hash_ull;
152 
153  // loops
154  for (atCol = 0; atCol < col; ++atCol) {
155  if (m_info[atCol]->getName() == colName) break;
156  }
157  if (atCol == col) [[unlikely]] { return tmp; }
158  for (int32_t r = 0; r < row; ++r) {
159  switch (m_data[r][atCol]->getType()) {
160  case cbMySQLType::String:
161  case cbMySQLType::DataTime:
162  case cbMySQLType::Date:
163  case cbMySQLType::Time: tmp[hash_string(m_data[r][atCol]->asString())] = r; break;
164  case cbMySQLType::Float: tmp[hash_float(m_data[r][atCol]->asFloat())] = r; break;
165  case cbMySQLType::Double: tmp[hash_double(m_data[r][atCol]->asFloat())] = r; break;
166  case cbMySQLType::Int: tmp[hash_int(m_data[r][atCol]->asInt())] = r; break;
167  case cbMySQLType::ULL: tmp[hash_ull(m_data[r][atCol]->asULL())] = r; break;
168  case cbMySQLType::Null: return tmp;
169  }
170  }
171  return tmp;
172 }
